cmake_minimum_required(VERSION 3.30)

project(ParaCL
    VERSION 5.1
)

set(PARACL_SRC_DIR ${PROJECT_SOURCE_DIR})

set(PARACL_INTERPRETER_SRC_DIR ${PARACL_SRC_DIR}/Interpreter)
set(PARACL_COMPILER_SRC_DIR ${PARACL_SRC_DIR}/Compiler)
set(PARACL_FRONTEND_SRC_DIR ${PARACL_SRC_DIR}/Frontend)

set(PARACL_INTERPRETER_BINARY_DIR ${CMAKE_CURRENT_BINARY_DIR}/subprojects/Interpreter)
set(PARACL_COMPILER_BINARY_DIR ${CMAKE_CURRENT_BINARY_DIR}/subprojects/Compiler)
set(PARACL_FRONTEND_BINARY_DIR ${CMAKE_CURRENT_BINARY_DIR}/subprojects/Frontend)

set(PARACL_INTERPRETER_EXE paracli)
set(PARACL_COMPILER_EXE paraclc)
set(PARACL_FRONTEND_EXE paraclf)

include(ExternalProject)

# build frontend
ExternalProject_Add(
    ParaCL-Frontend
    BINARY_DIR ${PARACL_FRONTEND_BINARY_DIR}
    SOURCE_DIR ${PARACL_FRONTEND_SRC_DIR}
    BUILD_COMMAND ${CMAKE_COMMAND} --build <BINARY_DIR>
    # TEST_COMMAND ${CMAKE_CTEST_COMMAND} --test-dir <BINARY_DIR>
    INSTALL_COMMAND ${CMAKE_COMMAND} -E copy "${PARACL_FRONTEND_BINARY_DIR}/paracl-front" "${CMAKE_BINARY_DIR}/${PARACL_FRONTEND_EXE}"
    CMAKE_ARGS -DCMAKE_CXX_COMPILER=${CMAKE_CXX_COMPILER} -DCMAKE_BUILD_TYPE=${CMAKE_BUILD_TYPE} -DCMAKE_TOOLCHAIN_FILE=${CMAKE_TOOLCHAIN_FILE}
    BUILD_ALWAYS ON
)

ExternalProject_Add(
    ParaCL-Compiler
    BINARY_DIR ${PARACL_COMPILER_BINARY_DIR}
    SOURCE_DIR ${PARACL_COMPILER_SRC_DIR}
    BUILD_COMMAND ${CMAKE_COMMAND} --build <BINARY_DIR>
    # TEST_COMMAND ${CMAKE_CTEST_COMMAND} --test-dir <BINARY_DIR>
    INSTALL_COMMAND ${CMAKE_COMMAND} -E copy ${PARACL_COMPILER_BINARY_DIR}/paracl ${CMAKE_CURRENT_BINARY_DIR}/${PARACL_COMPILER_EXE}
    CMAKE_ARGS -DCMAKE_CXX_COMPILER=${CMAKE_CXX_COMPILER} -DCMAKE_BUILD_TYPE=${CMAKE_BUILD_TYPE} -DPARACL_FRONT=${CMAKE_BINARY_DIR}/${PARACL_FRONTEND_EXE} #-DCMAKE_TOOLCHAIN_FILE=${CMAKE_TOOLCHAIN_FILE}
    BUILD_ALWAYS ON
)

ExternalProject_Add(
    ParaCL-Interpreter
    BINARY_DIR ${PARACL_INTERPRETER_BINARY_DIR}
    SOURCE_DIR ${PARACL_INTERPRETER_SRC_DIR}
    BUILD_COMMAND ${CMAKE_COMMAND} --build <BINARY_DIR>
    # TEST_COMMAND ${CMAKE_CTEST_COMMAND} --test-dir <BINARY_DIR>
    INSTALL_COMMAND ${CMAKE_COMMAND} -E copy ${PARACL_INTERPRETER_BINARY_DIR}/paracl ${CMAKE_CURRENT_BINARY_DIR}/${PARACL_INTERPRETER_EXE}
    CMAKE_ARGS -DCMAKE_CXX_COMPILER=${CMAKE_CXX_COMPILER} -DCMAKE_BUILD_TYPE=${CMAKE_BUILD_TYPE} -DPARACL_FRONT=${CMAKE_BINARY_DIR}/${PARACL_FRONTEND_EXE} #-DCMAKE_TOOLCHAIN_FILE=${CMAKE_TOOLCHAIN_FILE}
    BUILD_ALWAYS ON
)

# TESTS
include(CTest)
enable_testing()

add_test(
    NAME CompilerTests
    COMMAND ${CMAKE_CTEST_COMMAND} --test-dir ${PARACL_COMPILER_BINARY_DIR}
    LABELS ParaCL-Compiler
)

add_test(
    NAME InterpreterTests
    COMMAND ${CMAKE_CTEST_COMMAND} --test-dir ${PARACL_INTERPRETER_BINARY_DIR}
    LABELS ParaCL-Interpreter
)

# BENCHMARK
set(BENCHMARK_DIR ${CMAKE_CURRENT_SOURCE_DIR}/benchmark)
set(BENCMARK_RUNNER_IN ${BENCHMARK_DIR}/benchmark.sh.in)
set(BENCMARK_RUNNER ${CMAKE_CURRENT_BINARY_DIR}/benchmark)

set(PARACL_BENCMARK_DAT_DIR ${BENCHMARK_DIR}/dat)
set(PARACL_BENCMARK_TMP_DIR ${CMAKE_CURRENT_BINARY_DIR})

configure_file(
    "${BENCMARK_RUNNER_IN}"
    "${BENCMARK_RUNNER}"
    @ONLY
)

file(
    CHMOD "${BENCMARK_RUNNER}"
    PERMISSIONS OWNER_EXECUTE OWNER_READ OWNER_WRITE
)
